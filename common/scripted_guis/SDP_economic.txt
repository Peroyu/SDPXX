scripted_gui = {
	SDP_economic ={
		window_name = "SDP_economic_window"
		context_type = Player_context
		parent_window_token = politics_tab
		visible = {
			always = yes
		}
		dynamic_lists = {

			Contenders_List = {
				array = KMT_economic_sphere
				index = i
				change_scope = yes

				entry_container = diplomacy_country_list_country_entry
			}

		}
		effects = {
			enter_SDP_economic_window_button_click = {
				if = {
				   limit = {
				     NOT = { has_country_flag = open_SDP_economic_window_main }
					}
					set_country_flag = open_SDP_economic_window_main
				}
				else = {
				    clr_country_flag = open_SDP_economic_window_main
				}
			}
		}
	}
	properties = {
			Contenders_Entry_Flag = { image = "[This.GetFlag]" }

		}
	SDP_economic_main ={
		window_name = "SDP_economic_window_main"
		context_type = Player_context
		visible = {
			has_country_flag = open_SDP_economic_window_main
		}
	
	    effects = {
			enter_SDP_economic_window_close_button_click = {
				clr_country_flag = open_SDP_economic_window_main
			}
	    }

		properties = {
			SDP_economic_sphere_icon = {
				image = "[ROOT.GeteconomicsphereBackground]"
			}
		}
   }
