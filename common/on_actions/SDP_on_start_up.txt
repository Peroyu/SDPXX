			every_country = {

				multiply_variable = { Economic_civilian_num = num_of_civilian_factories }
				multiply_variable = { Economic_millita_num = num_of_military_factories }
				set_variable = {  Economic_num = Economic_civilian_num }
				add_to_variable = { Economic_num = Economic_millita_num }
			}
			every_country = {
				limit = {
					is_ai = no
				}
				country_event = {

				}

				    id = LaAme.1
					days = 400

				}
			}
				
			
		}
	}
